function loadShareholderData(user) { if (typeof POLICY_DATA === 'undefined') return; // البحث عن المساهم باستخدام الإيميل أو المعرف const shareholder = POLICY_DATA.shareholders.find(s => s.id === user.id || s.email === user.email); if (shareholder) { // تعبئة البيانات العلوية document.getElementById('sharesCount').innerText = shareholder.shares.toLocaleString(); document.getElementById('sharesPercent').innerText = shareholder.percent + '%'; setTimeout(() => { document.getElementById('percentBar').style.width = shareholder.percent + '%'; }, 500); const totalValue = shareholder.shares * 10; document.getElementById('sharesValue').innerText = totalValue.toLocaleString() + ' ر.س'; // تعبئة الجدول const tbody = document.getElementById('shareDetailsBody'); tbody.innerHTML = ` ${shareholder.name} أسهم عادية (Ordinary Shares) ${shareholder.shares.toLocaleString()} ${shareholder.percent}% نشطة وموثقة `; } }
